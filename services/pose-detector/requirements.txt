# services/pose-detector/requirements.txt
#
# Dependencies for the YOLO pose detection service.
# ultralytics pulls in PyTorch + CUDA support automatically.

# YOLOv8 — includes model loading, inference, and auto-download of weights.
# PyTorch with CUDA is pulled in as a dependency.
ultralytics>=8.3.0

# Redis client for reading frames and publishing detections.
redis==5.2.1

# async_timeout is required by redis on Python < 3.12.
# Not auto-installed in the CUDA base image due to system Python packaging.
async-timeout>=4.0.0

# NumPy for frame decoding (JPEG bytes → numpy array).
numpy>=1.24.0

# OpenCV headless for image decoding.
opencv-python-headless==4.10.0.84
